        tmp R0
        
        LDI tmp, 1
        minus_1 R31
        SUB minus_1, tmp
        
        infinity_roots R30
        MOV infinity_roots, minus_1
        
        a R1
        IN a
        
        b R2
        IN b
        
        CALL line_solve
        END

line_solve:
        tmp R0
        LDI tmp, 0
        
        num_of_roots R3
        x R4
        
        CP a, tmp
        BRNE a_is_not_0
            ; a == 0
            CP b, tmp
            BRNE b_is_not_0
                ; b == 0
                MOV num_of_roots, infinity_roots
                OUT num_of_roots
                RET
b_is_not_0:
            LDI num_of_roots, 0
            OUT num_of_roots
            RET
a_is_not_0:
        PUSH a
        PUSH b
        DIV
        PUSH minus_1
        MUL
        POP x
        
        LDI num_of_roots, 1
        OUT num_of_roots
        OUT x
        RET
